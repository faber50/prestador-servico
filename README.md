Configurar os dados do MySQL no .env
- DB_CONNECTION=mysql
- DB_HOST=127.0.0.1
- DB_PORT=3306
- DB_DATABASE=prestador_servico
- DB_USERNAME=root
- DB_PASSWORD=

Executar os comandos
- composer install
- npm install
- npm run production
- php artisan migrate
- php artisan serve
